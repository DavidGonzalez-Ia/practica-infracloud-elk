# Kubernetes Secrets Configuration Template
# Este archivo es una PLANTILLA. Nunca commits valores reales en Git.
# Para crear secrets reales, ejecuta estos comandos manualmente:

# 1. MySQL Secrets
kubectl create secret generic mysql-secret \
  --from-literal=MYSQL_ROOT_PASSWORD=rootpassword \
  --from-literal=MYSQL_DATABASE=taskdb \
  -n cloudedu \
  --dry-run=client -o yaml > mysql-secret.yaml

# 2. Elasticsearch Secrets (para producción)
kubectl create secret generic elasticsearch-secret \
  --from-literal=elastic-username=elastic \
  --from-literal=elastic-password=changeme \
  -n cloudedu \
  --dry-run=client -o yaml > elasticsearch-secret.yaml

# 3. Kibana Secrets (para producción)
kubectl create secret generic kibana-secret \
  --from-literal=elasticsearch-username=elastic \
  --from-literal=elasticsearch-password=changeme \
  -n cloudedu \
  --dry-run=client -o yaml > kibana-secret.yaml

# Para aplicar un secret:
# kubectl apply -f mysql-secret.yaml
# kubectl apply -f elasticsearch-secret.yaml
# kubectl apply -f kibana-secret.yaml

# Ver secrets:
# kubectl get secrets -n cloudedu
# kubectl describe secret mysql-secret -n cloudedu
